HelloYou can find a free example in the CODESYS Store, called "Visalization Dialog Box"

Code:VAR   dlgManager    : VisuElems.IDialogManager; (* Dialog manager *)   visuDialogConfirm : VisuElems.IVisualisationDialog; (* Reference to the custom dialog *)   visuDialogResult   : VisuElems.Visu_DialogResult; (* Get the button clicked. Possible results are OK, Cancel, Retry etc. *)END_VARdlgManager := VisuElems.g_VisuManager.GetDialogManager();visuDialogConfirm := dlgManager.GetDialog("NameOfDialog"); // Get the reference to the custom confirm dialogvisuDialogResult := visuDialogConfirm.GetResult();
